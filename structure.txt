CalmAI/
├── backend/                  # FastAPI logic (script generation, TTS, mixing)
│   ├── script_generator.py
│   ├── tts_engine.py
│   ├── mixer.py
│   ├── output/              # Generated audio files (optional)
│   └── main.py              # FastAPI entry point
├── calm_backend/            # Django project
│   ├── calm_backend/        # Django settings and URLs
│   ├── meditation/          # Django app with views and routes
│   └── manage.py
├── static/                  # Background audio assets
├── backgrounds/             # e.g., forest.mp3, ocean.mp3
├── requirements.txt         # Python dependencies
├── .env                     # API keys (optional — see below)
├── README.md                # Project overview
└── .gitignore               # Git exclusions

# Create virtual environment
python -m venv venv
source venv/bin/activate  # or venv\Scripts\activate on Windows

# Install dependencies
pip install -r requirements.txt

# Run FastAPI backend
uvicorn backend.main:app --reload

# Run Django server
cd calm_backend
python manage.py runserver
